/*
se filtra manufacturer menor o igual a 10 y id de model menor o igual a 10 where model id sea menor a 5
se imprime los resultados menores a 5
*/
select cmdl.manufacturer_id, cmfr.id 
from cat_model cmdl
inner join cat_manufacturer cmfr
on cmdl.manufacturer_id =  cmfr.id AND cmdl.manufacturer_id <=10 and  cmfr.id <=10
where cmdl.id <5

select  cmdl.nombre, cmfr.nombre 
from cat_model cmdl
inner join cat_manufacturer cmfr
on cmdl.manufacturer_id =  cmfr.id
where cmdl.id <20 order by cmdl.nombre
